# Solution
```python
s, n = input().split(), int(input())
last_list = []
if len(s) % n != 0:
    last = len(s) % n
    for i in range(1, last + 1):
        last_list.append(s[i - last - 1])


def group(iterable, count):
    return zip(*[iter(iterable)] * count)


a = list(group(s, n))
x = []
for i in a:
    x.append(list(i))
if len(s) % n != 0:
    x.append(last_list)
print(x)
```
