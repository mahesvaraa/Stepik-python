# –æ–±—ä—è–≤–ª–µ–Ω–∏–µ —Ñ—É–Ω–∫—Ü–∏–∏
def is_password_good(password):
    import re
    return all([re.findall(r'\d', password), re.findall(r'[A-Z]', password), re.findall(r'[a-z]', password), len(password) >= 8])

# —Å—á–∏—Ç—ã–≤–∞–µ–º –¥–∞–Ω–Ω—ã–µ
txt = input()

# –≤—ã–∑—ã–≤–∞–µ–º —Ñ—É–Ω–∫—Ü–∏—é
print(is_password_good(txt))
