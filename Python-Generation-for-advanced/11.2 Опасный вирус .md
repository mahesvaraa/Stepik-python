kod = {'execute': 'X', 'read': 'R', 'write': 'W'}
n = int(input())
a = []
for i in range(n):
    a.append(input().split())
    a[i] = {a[i][0]: a[i][1::]}
m = int(input())
b = []

for i in range(m):
    b.append(input().split())
    b[i][0] = kod.get(b[i][0])

for i in b:
    for j in a:
        if j.get(i[1]) is not None:
            print(['Access denied', "OK"][i[0] in j.get(i[1])])
