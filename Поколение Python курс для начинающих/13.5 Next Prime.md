# Solution
```python
# –æ–±—ä—è–≤–ª–µ–Ω–∏–µ —Ñ—É–Ω–∫—Ü–∏–∏
def is_prime(n):
    d = 2
    while n % d != 0:
        if n == 1:
            break
        d += 1
    return d == n


def get_next_prime(num):
    num += 1
    if not is_prime(num):
        return get_next_prime(num)
    else:
        return num


n = abs(int(input()))

# –≤—ã–∑—ã–≤–∞–µ–º —Ñ—É–Ω–∫—Ü–∏—é
print(get_next_prime(n))
```
