# Solution

```python
import numpy as np

n, m = map(int, input().split())
b = np.zeros((n, m))
q = 1
for i in range(n * m):
    for j in range(i + 1):
        if (i - j < m) and (j < n):
            b[j][m - (i - j) - 1] = q
            q += 1
for i in b:
    for j in i[::-1]:
        j = str(int(j))
        print(j.ljust(3), end=" ")
    print()
```
